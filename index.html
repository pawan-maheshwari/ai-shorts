const CACHE_KEY = 'aishorts_news_cache';

const CACHE_EXPIRY = 4 * 60 * 60 * 1000; // 4 hours

 

function saveToCache(items) {

  try {

    localStorage.setItem(CACHE_KEY, JSON.stringify({items, timestamp: Date.now()}));

  } catch(e) {}

}

 

function loadFromCache() {

  try {

    const c = localStorage.getItem(CACHE_KEY);

    if (!c) return null;

    const {items, timestamp} = JSON.parse(c);

    if (Date.now() - timestamp > CACHE_EXPIRY) {

      localStorage.removeItem(CACHE_KEY);

      return null;

    }

    return items;

  } catch(e) { return null; }

}

// Fixed HN (search_by_date)

async function fetchHN(){

  const url = "https://hn.algolia.com/api/v1/search_by_date?query=AI%20OR%20LLM%20OR%20GPT%20OR%20Claude%20OR%20Gemini%20OR%20xAI&tags=story&numericFilters=points>40&hitsPerPage=40";

  const res = await fetchT(url, {}, 10000);

  if(!res.ok) throw new Error("HN HTTP " + res.status);

  const data = await res.json();

  const kw = /\b(ai|llm|gpt|claude|gemini|xai|anthropic|openai)\b/i;

  const seen = new Set(), out = [];

  for(const h of (data.hits || [])){

    if(!h.title || !h.url || !kw.test(h.title)) continue;

    const key = h.title.toLowerCase(); if(seen.has(key)) continue; seen.add(key);

    let host = "source"; try{host = new URL(h.url).hostname.replace(/^www\./,"");}catch(e){}

    out.push({

      headline:h.title,

      summary:⁠ Trending on HN — ${h.points||0} pts, ${h.num_comments||0} comments. ⁠,

      source:host, url:h.url,

      category:guessCat(h.title), region:"Global",

      date:new Date().toISOString().slice(0,10),

      note:"via Hacker News • today"

    });

    if(out.length >= 12) break;

  }

  return out;

}
